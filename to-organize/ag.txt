## Install ##
#############

# From https://github.com/ggreer/the_silver_searcher/wiki/Windows

# If you don't have apt-cyg
wget rawgit.com/transcode-open/apt-cyg/master/apt-cyg
install apt-cyg /bin
# then
apt-cyg install automake pkg-config gcc-core libpcre-devel liblzma-devel
./build.sh && make install

############
## Search ##
############

# Search for 'mytextstring', ignore case, files of type .sas, 
#   recursively beginning in the current folder 
ag 'mytextstring' -i -G .sas > ~/ag_results.txt
